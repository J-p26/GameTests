function moveAISquare() { if (blackSquareIndex === null || redSquareIndex === null) return; const neighbors = getNeighbors(redSquareIndex); let bestMove = null; let minDistance = Infinity; // Find the best move for (const neighbor of neighbors) { // Allow moving onto the black square (to trigger game over) if (grid[neighbor] === null || grid[neighbor] === 'black') { const distance = getDistance(neighbor, blackSquareIndex); if (distance < minDistance) { minDistance = distance; bestMove = neighbor; } else if (distance === minDistance) { // Tie-breaker: prefer rightmost move const currentCol = redSquareIndex % gridSize; const neighborCol = neighbor % gridSize; if (neighborCol > currentCol) bestMove = neighbor; } } } // Move to the best neighbor if (bestMove !== null) { grid[redSquareIndex] = null; grid[bestMove] = 'red'; redSquareIndex = bestMove; // Increment move count and update speed redSquareMoveCount++; if (redSquareMoveCount % 20 === 0 && aiIntervalTime > 100) { aiIntervalTime -= 100; // Decrease interval by 0.1s (100ms) clearInterval(aiInterval); aiInterval = setInterval(moveAISquare, aiIntervalTime); } speedInfo.textContent = `Red Square Moves: ${redSquareMoveCount} | Speed: ${(aiIntervalTime / 1000).toFixed(1)}s`; // Check for collision (red square catches black square) if (redSquareIndex === blackSquareIndex) { clearInterval(aiInterval); gameOver(); return; } updateGrid(); } }
